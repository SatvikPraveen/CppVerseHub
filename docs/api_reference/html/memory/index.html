// Custom pool allocator for entities template<typename T, size_t PoolSize> class PoolAllocator { private: alignas(T) char pool[PoolSize * sizeof(T)]; std::bitset<PoolSize> used; size_t nextFree = 0; public: T* allocate() { for (size_t i = 0; i < PoolSize; ++i) { size_t index = (nextFree + i) % PoolSize; if (!used[index]) { used[index] = true; nextFree = (index + 1) % PoolSize; return reinterpret_cast<T*>(&pool[index * sizeof(T)]); } } return nullptr; } };
